 38 BigNum::BigNum(const int b)     //将一个int类型的变量转化为大数
 39 { 
 40     int c,d = b;
 41     len = 0;
 42     memset(a,0,sizeof(a));
 43     while(d > MAXN)
 44     {
 45         c = d - (d / (MAXN + 1)) * (MAXN + 1); 
 46         d = d / (MAXN + 1);
 47         a[len++] = c;
 48     }

 51 BigNum::BigNum(const char*s)     //将一个字符串类型的变量转化为大数
